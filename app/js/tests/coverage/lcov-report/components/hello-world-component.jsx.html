import React, { Component } from 'react';
const PropTypes = React.PropTypes;
 
class Root extends Component {
 
    getUser() {
        this.props.getUser(this.textInput.value);
    }
 
    render() {
        return (
            <div>
                <h1 className="greeting" id="greeting">Hello user...</h1>
                <h1 className="user" id="user">{this.props.user}</h1>
                <label>GitHub User Number: <input ref={(input) => { this.textInput = input; }} /></label>
                <button onClick={this.getUser.bind(this)}>Get User</button>
            </div>
        );
    }
}
 
Root.propTypes = {
    getUser: PropTypes.func.isRequired,
    user: PropTypes.string
};
 
export default Root;
